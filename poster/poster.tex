\begin{column}{\colwidth}

  \begin{block}{Motivation \& Problem}

    Centralized exchanges (CEXs) provide deep liquidity and low-latency orderbooks, while
    decentralized exchanges (DEXs) execute on-chain via AMMs with self-custody and transparency.
    These structural differences create measurable gaps in \textbf{price discovery}, \textbf{execution quality},
    and \textbf{transaction costs}.

    \vspace{0.6em}
    \textbf{Goal:} Build an end-to-end system that monitors CEX and DEX venues in real time and quantifies:
    deviation dynamics, lead--lag behavior, and slippage characteristics.

  \end{block}

  \begin{block}{Data Sources \& Scope}

    \textbf{CEX feeds (WebSocket):} Coinbase, MEXC, Binance, Gate.io, OKX, Bybit, HTX, KuCoin.

    \vspace{0.4em}
    \textbf{DEX feeds (on-chain events):} Uniswap V2 and Uniswap V3 Swap events.

    \begin{itemize}
      \item \textbf{Chains used in analysis:} Ethereum + BSC
      \item \textbf{Pairs:} BTC/USDT, ETH/USDT, BNB/USDT
      \item \textbf{DEX listener capability:} multi-chain configurable (additional networks supported via configuration)
    \end{itemize}

    \vspace{0.4em}
    \textbf{Dataset size (placeholder):} \#CEX updates = \underline{\hspace{2cm}}, \#DEX swaps = \underline{\hspace{2cm}}, period = \underline{\hspace{2cm}}.

  \end{block}

  \begin{alertblock}{Contributions}

    \begin{itemize}
      \item \textbf{Multi-exchange CEX price index} (volume-weighted reference price)
      \item \textbf{Real-time DEX swap capture} (Uniswap V2/V3 on Ethereum/BSC)
      \item \textbf{Deviation \& lead--lag analysis} between CEX index and DEX execution

  \begin{block}{Methodology (Core Metrics)}

    \textbf{CEX reference price (index):}
    \[
      \text{price\_index}(t)=\frac{\sum_i \text{mid}_i(t)\cdot \text{vol}_i}{\sum_i \text{vol}_i},
      \quad \text{mid}_i(t)=\frac{\text{bid}_i(t)+\text{ask}_i(t)}{2}
    \]

    \textbf{Deviation per DEX trade:}
    \[
      \text{deviation}=\frac{\text{DEX\_price}-\text{CEX\_index}}{\text{CEX\_index}}\times 100\%
    \]

    \textbf{Lead--lag:} cross-correlation of log returns over lags $\tau \in [-60s, +60s]$.

    \textbf{Slippage (V2 theoretical):}
    \[
      \Delta y = y-\frac{xy}{x+\Delta x}, \quad
      \text{slippage}=\left|1-\frac{\Delta y/\Delta x}{y/x}\right|
    \]

  \end{block}

  \begin{block}{Implementation}

      \item \textbf{Analysis:} Python (\texttt{analysis/})
      \item \textbf{Dashboard:} Streamlit (\texttt{price-index/dashboard.py})
      \item \textbf{Analysis runner:} \texttt{analysis/run\_analysis.py} (deviation, lead--lag, slippage, volume)
    \end{itemize}
